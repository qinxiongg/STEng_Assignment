// cURL (Windows 11)

// cURL for create task
$body = @{
    username = "PL"
    password = "abc123!!"
    appAcronym = "App1"
    taskName = "test1"
    taskDescription = ""
    taskNotes = ""
    taskPlan = "plan"
}

$response = Invoke-WebRequest -Uri "http://localhost:3000/api/demo/CreateTask" `
-Method Post `
-ContentType "application/json" `
-Body ($body | ConvertTo-Json)

# Output the full response
$response
